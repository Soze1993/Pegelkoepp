       const bcrypt = require('bcryptjs');
       const router = Router();

       router.post('/login', async (req, res) => {
         const { pin } = req.body || {};
         const hash = process.env.PIN_HASH;
         if (!pin || !hash) return res.status(400).json({ error: 'Bad request' });
         const ok = await bcrypt.compare(String(pin), hash);
         if (!ok) return res.status(401).json({ error: 'Falscher PIN' });
         req.session.regenerate((err) => {

           req.session.authenticated = true;
           res.json({ ok: true });
         });
       });

       router.post('/logout', (req, res) => {
         req.session.destroy(() => res.json({ ok: true }));
       });

       router.get('/status', (req, res) => {
         res.json({ authenticated: !!(req.session && req.session.authenticated) });
       });

       module.exports = router;

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /api/auth/login | Untrusted PIN submission; treated as the only path to elevate trust |
| Browser → /api/players (write) | Untrusted body; must validate types and reject on missing fields |
| Session cookie → server | Cookie tampering attempts; mitigated by express-session signing |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-07 | Spoofing | PIN brute force | accept | bcrypt cost 10 (~100ms/attempt) + single shared PIN for a private club app — RESEARCH.md Security Domain notes the club accepts this risk. No rate limiting in Phase 1; add in Phase 5 if needed. |
| T-01-08 | Elevation of Privilege | Session fixation | mitigate | `req.session.regenerate()` is called on successful login (Task 1 step 3); verified by test R8 comparing pre/post cookie values |
| T-01-09 | Tampering | Unauthenticated write to /api/players | mitigate | `requireSession` middleware applied to POST, PUT, PUT-archive (Task 2); verified by tests P1, P2, P3 |
| T-01-10 | Tampering | SQL injection via name/emoji | mitigate | All UPDATE/INSERT use `db.prepare(...).run(...)` with bound parameters (Task 2); no string interpolation |
| T-01-11 | Information Disclosure | bcrypt error message leakage | mitigate | Auth route returns the same generic German `Falscher PIN` for any failure path (wrong pin OR missing hash → both 401/400 with generic text), no internal state revealed |
| T-01-12 | Repudiation | No audit log of who-did-what | accept | Shared PIN means all writes are attributable only to "the club"; per-user accounts are explicitly out of scope (REQUIREMENTS.md Out of Scope) |
| T-01-13 | Tampering | Type confusion on POST body (e.g., name: 123 or name: { sql: '...' }) | mitigate | `typeof name !== 'string'` check returns 400 before any DB query (Task 2); verified by test P7 |
</threat_model>
