async function submitThrow(player_id, throw_index, value, meta) {
  var res = await fetch('/api/games/' + S.aktSpiel.gameId + '/throws', {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({ player_id: player_id, throw_index: throw_index, value: value, meta: meta || undefined })
  });
  if (!res.ok) {
    if (res.status === 409) { notify('Doppelter Wurf', 'var(--red)'); } else { notify('Fehler beim Wurf', 'var(--red)'); }
    return null;
  }
  return await res.json();
}

async function submitUndo() {
  var res = await fetch('/api/games/' + S.aktSpiel.gameId + '/undo', { method: 'POST', headers: { 'content-type': 'application/json' } });
  if (!res.ok) { notify('Kein Wurf zum Rueckgaengig machen', 'var(--red)'); return; }
  var data = await res.json();
  S.aktSpiel.state = data.state;
  renderSpielenTab();
}

async function doNWurf() {
  if (!S.aktSpiel) return;
  var state = S.aktSpiel.state;
  var stid = S.aktSpiel.type_key;
  var akt = state.players[state.aktSpIdx];

  if (stid === 'dreiVollen' || stid === 'plusMinus') {
    if (selK === null) { notify('Bitte waehlen', 'var(--red)'); return; }
    var throwIndex = akt.wuerfe ? akt.wuerfe.length : 0;
    var value = selK;
    var meta;
    if (stid === 'plusMinus') {
      var pmR = state.pmRunde || 1;
      var wi = pmR - 1;
      if (selK === 0) { if (wi === 2) value = 9; else if (wi === 3 || wi === 4) value = 1; else value = 0; }
    }
    selK = null;
    var data = await submitThrow(akt.id, throwIndex, value, meta);
    if (!data) return;
    S.aktSpiel.state = data.state;
    if (data.finished) { handleGameFinished(data.state); } else { renderSpielenTab(); }
    return;
  }

  if (stid === 'grosseHaus' || stid === 'kleineHaus') {
    if (hausWurf === null) { notify('Ergebnis waehlen', 'var(--red)'); return; }
    if (!hausSlot) { notify('Stelle waehlen', 'var(--red)'); return; }
    var throwIndex2 = akt.wuerfe ? akt.wuerfe.length : 0;
    var val2 = hausWurf;
    if (stid === 'kleineHaus' && hausWurf === 0) val2 = 9;
    var meta2 = { slot: hausSlot };
    hausWurf = null; hausSlot = null;
    var data2 = await submitThrow(akt.id, throwIndex2, val2, meta2);
    if (!data2) return;
    S.aktSpiel.state = data2.state;
    if (data2.finished) { handleGameFinished(data2.state); } else { renderSpielenTab(); }
  }
}
