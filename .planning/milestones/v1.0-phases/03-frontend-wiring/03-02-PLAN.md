3. SOCKET.IO CLIENT + CONNECTION DOT + GAME EVENTS (per D-07, D-10, D-11, D-12):
```
function initSocket() {
  socket = io();

  var connDot = document.getElementById('connDot');
  socket.on('connect', function() { if (connDot) connDot.className = 'conn-dot green'; if (S.aktSpiel) socket.emit('join', S.aktSpiel.gameId); });
  socket.on('disconnect', function() { if (connDot) connDot.className = 'conn-dot red'; });

  socket.on('game:started', function(data) {
    S.aktSpiel = { gameId: data.gameId, state: data.state, type_key: data.type_key };
    socket.emit('join', data.gameId);
    showTab('spielen');
  });

  socket.on('throw:applied', function(data) {
    if (S.aktSpiel) S.aktSpiel.state = data.state;
    renderSpielenTab();
  });

  socket.on('undo:applied', function(data) {
    if (S.aktSpiel) S.aktSpiel.state = data.state;
    renderSpielenTab();
  });

  socket.on('game:finished', function(data) {
    if (S.aktSpiel && S.aktSpiel._finishing) return;
    showWinnerBanner(data.state);
  });
}

function joinGameRoom(gameId) {
  if (socket) socket.emit('join', gameId);
}
```

function getWinnerName(state, type_key) {
  if (!state) return 'Unbekannt';
  if (type_key === 'kda') { return (state.gewinner && state.gewinner.name) ? state.gewinner.name : 'Unbekannt'; }
  if (type_key === 'fuchsjagd') {
    if (!state.winner) return state.fuchs ? state.fuchs.name : 'Unbekannt';
    return state.winner === 'fuchs' ? (state.fuchs ? state.fuchs.name : 'Fuchs') : (state.sieger ? state.sieger.name : 'Jaeger');
  }
  if (type_key === 'viergewinnt') {
    if (state.winner === 'draw') return 'Unentschieden';
    if (state.winner === 'X') { return state.tX && state.tX.length ? state.tX.map(function(p){return p.name;}).join(' & ') : 'Team X'; }
    if (state.winner === 'O') { return state.tO && state.tO.length ? state.tO.map(function(p){return p.name;}).join(' & ') : 'Team O'; }
    return 'Unbekannt';
  }
  if (type_key === 'bilderkegel') { return (state.gewinner && state.gewinner.name) ? state.gewinner.name : 'Unbekannt'; }
  // dreiVollen, plusMinus, grosseHaus, kleineHaus, anker — all have state.players sorted by score
  if (state.players && state.players.length) {
    var pl = state.players.slice();
    if (type_key === 'kleineHaus') {
      pl.sort(function(a,b){ var va=hn(a.slots)||999,vb=hn(b.slots)||999; return va-vb; });
    } else {
      pl.sort(function(a,b){ return totPts(b, type_key) - totPts(a, type_key); });
    }
    return pl[0].name || 'Unbekannt';
  }
  return 'Unbekannt';
}
