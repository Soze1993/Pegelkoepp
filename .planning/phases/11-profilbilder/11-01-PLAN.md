In `server/routes/players.js`, add the following at the top of the file, after the existing `require` lines:

```js
const express = require('express');
const fs = require('fs');
const path = require('path');
const UPLOADS_DIR = path.join(__dirname, '../../public/uploads/profiles');
const rawUpload = express.raw({ type: ['image/jpeg', 'image/png'], limit: '5mb' });
```

Then add the upload route BEFORE the `module.exports = router;` line at the bottom:

```js
// POST /api/players/:id/photo — upload profile photo (D-04, D-05, D-06, D-07)
router.post('/:id/photo', requireSession, rawUpload, (req, res) => {
  const id = Number(req.params.id);
  if (!Number.isInteger(id) || id <= 0) {
    return res.status(400).json({ error: 'invalid id' });
  }
  // Verify player exists and is not archived
  const player = db.prepare('SELECT id FROM players WHERE id = ? AND archived = 0').get(id);
  if (!player) return res.status(404).json({ error: 'player not found' });

  const buf = req.body;
  if (!Buffer.isBuffer(buf) || buf.length === 0) {
    return res.status(400).json({ error: 'body must be image/jpeg or image/png' });
  }
  // Magic byte validation (D-05 security, RESEARCH.md Pattern 1)
  const isJpeg = buf[0] === 0xff && buf[1] === 0xd8 && buf[2] === 0xff;
  const isPng  = buf.length >= 8 &&
    buf[0] === 0x89 && buf[1] === 0x50 && buf[2] === 0x4e && buf[3] === 0x47 &&
    buf[4] === 0x0d && buf[5] === 0x0a && buf[6] === 0x1a && buf[7] === 0x0a;
  if (!isJpeg && !isPng) {
    return res.status(400).json({ error: 'invalid image: not JPEG or PNG' });
  }
  // Always save as {id}.jpg regardless of input type (D-06)
  fs.writeFileSync(path.join(UPLOADS_DIR, id + '.jpg'), buf);
  res.json({ ok: true, url: '/uploads/profiles/' + id + '.jpg' });
});
```

Placement matters: the route uses `router.post('/:id/photo', ...)`. Because Express matches routes in declaration order, and the existing `router.put('/:id', ...)` and `router.put('/:id/archive', ...)` use PUT (not POST), there is no conflict. The only existing POST is `router.post('/', ...)` which is a different path. No reordering needed.

Do NOT add `express.raw()` to app.js or as `app.use('/api/players', ...)` — it must be inline on this route only (per D-05 and RESEARCH.md Pitfall 1).

After adding the route, verify the file syntax is valid with `node --check server/routes/players.js` before running tests.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → POST /api/players/:id/photo | Unauthenticated or authenticated browser/app sends raw binary body |
| disk write | Server writes received bytes to public/uploads/profiles/ |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-01 | Elevation of Privilege | POST /:id/photo — unauthenticated access | mitigate | requireSession FIRST in middleware chain, before rawUpload — 401 fires before body is read |
| T-11-02 | Tampering | Path traversal via :id parameter | mitigate | `Number(req.params.id)` + `isInteger` + `> 0` — result is always a plain integer, filename is `{integer}.jpg`, no `..` possible |
| T-11-03 | Tampering | Content-type spoofing (claim image/jpeg, send PHP/HTML) | mitigate | Magic byte validation checks first 3 bytes (JPEG) or 8 bytes (PNG) regardless of declared Content-Type |
| T-11-04 | Denial of Service | 5MB upload flood via concurrent requests | mitigate | express.raw `limit:'5mb'` — 413 returned before full body read; existing app.js error handler catches `entity.too.large` |
| T-11-05 | Tampering | Overwrite another player's photo by guessing id | accept | Only authenticated sessions (requireSession) can upload; club app with shared PIN; ids are not secret within the club |
| T-11-SC | Tampering | npm/pip install supply chain | accept | No new packages installed in this plan; express.raw is built into installed Express 5.2.1 |
</threat_model>
